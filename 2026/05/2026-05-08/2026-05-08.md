# [3629. 通过质数传送到达终点的最少跳跃次数](https://leetcode.cn/problems/minimum-jumps-to-reach-end-via-prime-teleportation/)

## spf & bfs

`spf[x]=p` : x 的最小质因子是 p 

`spf[x]==x` : x 是质数

---

## 代码实现 (C++)
```cpp
class Solution {
public:
    int minJumps(vector<int>& nums) {
        int n=nums.size();
        int M=0;
        for(int x:nums){
            M=max(M,x);
        }
        vector<int>spf(M+1);
        for(int i=2;i<=M;++i){
            spf[i]=i;
        }
        for(int i=2;i*i<=M;++i){
            if(spf[i]==i){
                for(int j=i*i;j<=M;j+=i){
                    if(spf[j]==j){
                        spf[j]=i;
                    }
                }
            }
        }

        vector gra(M+1,vector<int>());
        for(int i=0;i<n;++i){
            int x=nums[i];
            while(x>1){
                int p=spf[x];
                if(gra[p].empty()||gra[p].back()!=i){
                    gra[p].push_back(i);
                }
                while(x%p==0){
                    x/=p;
                }
            }
        }
        queue<int>qu;
        vector<int>dis(n,-1);
        vector<char>vis(M+1);
        qu.push(0);
        dis[0]=0;
        while(qu.size()){
            int u=qu.front();
            qu.pop();
            if(u==n-1){
                return dis[u];
            }
            if(u-1>=0&&dis[u-1]==-1){
                dis[u-1]=dis[u]+1;
                qu.push(u-1);
            }
            if(u+1<n&&dis[u+1]==-1){
                dis[u+1]=dis[u]+1;
                qu.push(u+1);
            }
            int x=nums[u];
            if(x>1&&spf[x]==x){ 
                int p=x;
                if(vis[p]==0) {
                    vis[p]=1;
                    for(int v:gra[p]){
                        if(dis[v]==-1){
                            dis[v]=dis[u]+1;
                            qu.push(v);
                        }
                    }
                }
            }
        }
        return -1;
    }
};
```

- 时间复杂度: $O(N\log N)$
- 空间复杂度: $O(M)$
