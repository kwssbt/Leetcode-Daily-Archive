## 代码实现 (C++)
```cpp
class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int n=nums.size();
        vector<int>d(limit*2+2);
        for(int i=0;i<n/2;++i){
            int x=nums[i];
            int y=nums[n-1-i];
            int l=min(x,y)+1;
            int r=max(x,y)+limit;
            d[l]--;
            d[x+y]--;
            d[x+y+1]++;
            d[r+1]++;
        }
        int ans=n;
        int sum=n;
        for(int i=2;i<=limit*2;++i){
            sum+=d[i];
            ans=min(ans,sum);
        }
        return ans;
    }
};
```

- 时间复杂度: $O(N+limit)$
- 空间复杂度: $O(limit)$
