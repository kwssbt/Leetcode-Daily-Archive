## 代码实现 (C++)
```cpp
class Solution {
public:
    int rotatedDigits(int n) {
        int ans=0;
        function<int(int)>re=[](int x)->int{
            if(x==0||x==1||x==8){
                return x;
            }
            if(x==2){
                return 5;
            }
            if(x==5){
                return 2;
            }
            if(x==6){
                return 9;
            }
            if(x==9){
                return 6;
            }
            return -1;
        };
        function<int(int)>check=[&](int x)->int{
            int t=x;
            int ret=0;
            int bit=1;
            while(x){
                int b=re(x%10);
                if(b==-1)return 0;
                ret+=b*bit;
                bit*=10;
                x/=10;
            }
            return ret!=t;
        };
        for(int i=1;i<=n;++i){
            ans+=check(i);
        }
        return ans;
    }
};
```

- 时间复杂度: $O(NB)$
- 空间复杂度: $O(1)$
