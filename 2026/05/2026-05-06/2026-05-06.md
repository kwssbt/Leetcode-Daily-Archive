## 代码实现 (C++)
```cpp
class Solution {
public:
    vector<vector<char>> rotateTheBox(vector<vector<char>>& box) {
        int n=box.size(),m=box[0].size();
        vector re(m,vector<char>(n));
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                re[j][n-i-1]=box[i][j];
            }
        }
        for(int i=m-2;i>=0;--i){
            for(int j=0;j<n;++j){
                if(re[i][j]!='#')continue;
                if(re[i+1][j]!='.')continue;
                int p=i+1;
                while(p<m&&re[p][j]=='.')++p;
                re[i][j]='.';
                re[p-1][j]='#';
            }
        }
        return re;
    }
};
```

- 时间复杂度: $O(N^3)$
- 空间复杂度: $O(1)$
