## 暴力

枚举以 $(x,y)$ 为中心，a 为内径的菱形

set 去重


---

## 代码实现 (C++)
```cpp
class Solution {
public:
    vector<int> getBiggestThree(vector<vector<int>>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        function<int(int,int,int)>get_sum=[&](int x,int y,int a)->int{
            if(a==0){
                return grid[x][y];
            }
            int sum=0;
            for(int i=0;i<a;++i)sum+=grid[x-a+i][y+i];
            for(int i=0;i<a;++i)sum+=grid[x+a-i][y-i];
            for(int i=0;i<a;++i)sum+=grid[x+i][y+a-i];
            for(int i=0;i<a;++i)sum+=grid[x-i][y-a+i];
            return sum;
        };
        set<int,greater<>>st;
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                for(int a=0;i-a>=0&&j-a>=0&&i+a<n&&j+a<m;++a){
                    st.insert(get_sum(i,j,a));
                }
            }
        }
        vector<int>ans;
        for(int x:st){
            ans.push_back(x);
            if(ans.size()==3){
                break;
            }
        }
        return ans;
    }
};
```

- 时间复杂度: $O(N^4)$
- 空间复杂度: $O(n^3)$

## 优化

set 中只存至多三个元素

```cpp
class Solution {
public:
    vector<int> getBiggestThree(vector<vector<int>>& grid) {
        int n=grid.size();
        int m=grid[0].size();
        function<int(int,int,int)>get_sum=[&](int x,int y,int a)->int{
            if(a==0){
                return grid[x][y];
            }
            int sum=0;
            for(int i=0;i<a;++i)sum+=grid[x-a+i][y+i];
            for(int i=0;i<a;++i)sum+=grid[x+a-i][y-i];
            for(int i=0;i<a;++i)sum+=grid[x+i][y+a-i];
            for(int i=0;i<a;++i)sum+=grid[x-i][y-a+i];
            return sum;
        };
        set<int,greater<>>st;
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                for(int a=0;i-a>=0&&j-a>=0&&i+a<n&&j+a<m;++a){
                    st.insert(get_sum(i,j,a));

                    if(st.size()>3){
                        st.erase(prev(st.end()));
                    }
                }
            }
        }
        return vector<int>(st.begin(),st.end());
    }
};
```

- 时间复杂度: $O(N^4)$
- 空间复杂度: $O(1)$
