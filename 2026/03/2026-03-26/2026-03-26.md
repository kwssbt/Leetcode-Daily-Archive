# [3548. 等和矩阵分割 II](https://leetcode.cn/problems/equal-sum-grid-partition-ii/)

**2026-03-26**

**题面难度：Hard** 

[查看PDF题面](./2026-03-26.png)

## 枚举 

### 问题简化

**只允许从第一部分删除元素，只能横着分割**

设矩阵总和为 $sum$

设第一部分元素和为 $cur$ 

则第二部分元素和为 $sum-cur$

若不删

有 $cur=sum-cur$

若第一部分删去 $x$ 

有 $cur-x=sum-cur$

即 $x=2\times cur-sum$

遍历矩阵一边维护 $cur$ 一边检查 $x$ 之前是否出现过

### 代码复用

删除第二部分元素：矩阵反转180度

竖着分割：矩阵反转90度，270度

## 代码实现 (C++)
```cpp
class Solution {
    vector<vector<int>>re90(vector<vector<int>>&a){
        int n=a.size();
        int m=a[0].size();
        vector b(m,vector<int>(n));
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                b[j][n-i-1]=a[i][j];
            }
        }
        return b;
    }
    vector<vector<int>>re180(vector<vector<int>>&a){
        int n=a.size();
        int m=a[0].size();
        vector b(n,vector<int>(m));
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                b[n-i-1][m-j-1]=a[i][j];
            }
        }
        return b;
    }
    vector<vector<int>>re270(vector<vector<int>>&a){
        int n=a.size();
        int m=a[0].size();
        vector b(m,vector<int>(n));
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                b[m-j-1][i]=a[i][j];
            }
        }
        return b;
    }

    long long sum;
    
    bool check(vector<vector<int>>grid){
        int n=grid.size();
        int m=grid[0].size();  

        if(n==1)return 0;//要求两部分非空

        //单独判断将第0行作为第一部分
        //第一部分是单条横线，为了保持联通，只能选择不删，删最左边，删最右边
        long long row0=0;
        for(int x:grid[0])row0+=x;
        if(row0==sum-row0||row0-grid[0][0]==sum-row0||row0-grid[0][m-1]==sum-row0)return 1;

        unordered_set<long long>st;
        st.insert(0);
        long long cur=0;
        for(int x:grid[0]){
            st.insert(x);
            cur+=x;
        }
        //最下面一行（第n-i行）必须属于第二部分
        for(int i=1;i<n-1;++i){
            for(int x:grid[i]){
                st.insert(x);
                cur+=x;
            }
            
            long long target=2*cur-sum;

            if(m==1){//第一部分是单条竖线，为了保持联通，只能选择不删，删顶部，删底部
                if(target==0||target==grid[0][0]||target==grid[i][0])return 1;
            }
            else{
                if(st.count(target))return 1;
            }
        }
        return 0;
    }
public:
    bool canPartitionGrid(vector<vector<int>>&grid){
        this->sum=0;
        for(auto&row:grid){
            for(int x:row){
                sum+=x;
            }
        }
        return check(grid)||check(re90(grid))||check(re180(grid))||check(re270(grid));
    }
};
```

- 时间复杂度: $O(NM)$
- 空间复杂度: $O(NM)$
