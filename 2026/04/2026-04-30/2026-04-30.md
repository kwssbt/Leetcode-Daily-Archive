## 代码实现 (C++)
```cpp
class Solution {
public:
    int maxPathScore(vector<vector<int>>& grid, int K) {
        int n=grid.size(),m=grid[0].size();

        vector dp(n+1,vector<vector<int>>(m+1,vector<int>(K+2,INT_MIN)));
        for(int k=1;k<=K+1;++k){
            dp[0][1][k]=0;
        }
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                for(int k=0;k<=K;++k){
                    int nk=grid[i][j]==0?k:k-1;
                    dp[i+1][j+1][k+1]=max(dp[i+1][j][nk+1],dp[i][j+1][nk+1])+grid[i][j];
                }
            }
        }
        return dp[n][m][K+1]<0?-1:dp[n][m][K+1];
    }
};
```

- 时间复杂度: $O(NMK)$
- 空间复杂度: $O(NMK)$
