## 代码实现 (C++)
```cpp
class Solution {
public:
    int minOperations(vector<vector<int>>& grid, int x) {
        int n=grid.size(),m=grid[0].size();
        vector<int>a;
        int k=grid[0][0]%x;
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                if(grid[i][j]%x!=k){
                    return -1;
                }
                a.push_back(grid[i][j]);
            }
        }
        ranges::sort(a);
        int tar=a[n*m/2];
        int ans=0;
        for(int v:a){
            ans+=abs(v-tar)/x;
        }
        return ans;
    }
};
```

- 时间复杂度: $O(NM)$
- 空间复杂度: $O(NM)$
