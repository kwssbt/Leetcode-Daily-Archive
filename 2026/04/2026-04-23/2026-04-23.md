## 代码实现 (C++)
```cpp
class Solution {
public:
    vector<long long> distance(vector<int>& nums) {
        int n=nums.size();
        vector<long long>arr(n);
        unordered_map<int,vector<int>>st;
        for(int i=0;i<n;++i){
            st[nums[i]].push_back(i);
        }
        for(auto [_,p]:st){
            int m=p.size();
            vector<long long>pre(m+1);
            for(int i=0;i<m;++i){
                pre[i+1]=pre[i]+p[i];
            }
            for(int i=1;i<=m;++i){
                long long L=1LL*i*p[i-1]-pre[i];
                long long R=pre[m]-pre[i]-1LL*(m-i)*p[i-1];
                arr[p[i-1]]=L+R;
            }
        }
        return arr;
    }
};
```

- 时间复杂度: $O(NM)$
- 空间复杂度: $O(N)$
