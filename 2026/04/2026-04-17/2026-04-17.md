## 代码实现 (C++)
```cpp
class Solution {
public:
    int minMirrorPairDistance(vector<int>& nums) {
        int n=nums.size();
        vector<int>re(n);
        for(int i=0;i<n;++i){
            int t=nums[i];
            int ret=0;
            while(t){
                ret=ret*10+t%10;
                t/=10;
            }
            re[i]=ret;
        }
        int ans=INT_MAX;
        unordered_map<int,int>st;
        for(int i=0;i<n;++i){
            if(st.count(nums[i])){
                ans=min(ans,i-st[nums[i]]);
            }
            st[re[i]]=i;
        }
        return ans==INT_MAX?-1:ans;
    }
};
```

- 时间复杂度: $O(N)$
- 空间复杂度: $O(N)$
