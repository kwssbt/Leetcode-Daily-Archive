## 代码实现 (C++)
```cpp
class Solution {
    typedef long long ll;
public:
    int maxDistance(int side, vector<vector<int>>& points, int k) {
        vector<ll>p;
        for(auto&point:points){
            ll x=point[0];
            ll y=point[1];
            if(y==0){
                p.push_back(x);
            }
            else if(x==side){
                p.push_back(side+y);
            }
            else if(y==side){
                p.push_back(3LL*side-x);
            }
            else{
                p.push_back(4LL*side-y);
            }
        }
        sort(p.begin(),p.end());

        function<bool(ll)>check=[&](ll ans)->bool{
            for(ll st:p){
                ll end=st+side*4LL-ans;
                ll cur=st;
                bool ok=1;
                for(int i=1;ok&&i<k;++i){
                    auto it=lower_bound(p.begin(),p.end(),cur+ans);
                    if(it==p.end()||*it>end){
                        ok=0;
                    }
                    else{
                        cur=*it;
                    }
                }
                if(ok){
                    return 1;
                }
            }
            return 0;
        };

        ll l=0,r=4LL*side/k,mid,ans=l;
        while(l<=r){
            mid=l+r>>1;
            if(check(mid))ans=mid,l=mid+1;
            else r=mid-1;
        }
        return ans;
    }
};
```

- 时间复杂度: $O(N^2\log N \log side)$
- 空间复杂度: $O(N)$
