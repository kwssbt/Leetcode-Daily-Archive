## 代码实现 (C++)
```cpp
class Solution {
public:
    int minimumDistance(vector<int>& nums) {
        int n=nums.size();

        int ans=INT_MAX;
        vector P(n+1,vector<int>());
        for(int i=0;i<n;++i){
            P[nums[i]].push_back(i);
        }
        for(auto&p:P){
            if(p.size()>=3){
                for(int i=0;i+2<p.size();++i){
                    int t=abs(p[i]-p[i+1])+abs(p[i+1]-p[i+2])+abs(p[i+2]-p[i]);
                    ans=min(ans,t);
                }
            }
        }
        return ans==INT_MAX?-1:ans;
    }
};
```

- 时间复杂度: $O(N)$
- 空间复杂度: $O(N)$
