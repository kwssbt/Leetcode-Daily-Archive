## bfs

四个相邻块中有两个及以上被访问过

---

## 代码实现 (C++)
```cpp
class Solution {
    vector<int>mo{-1,0,1,0,-1};
public:
    bool containsCycle(vector<vector<char>>& grid) {
        int n=grid.size(),m=grid[0].size();
        vector vis(n,vector<int>(m));
        function<bool(int,int)>bfs=[&](int x,int y)->bool{
            queue<pair<int,int>>q;
            q.push({x,y});
            while(q.size()){
                auto [x,y]=q.front();
                q.pop();
                int cnt=0;
                for(int i=0;i<4;++i){
                    int nx=x+mo[i];
                    int ny=y+mo[i+1];
                    if(nx>=n||ny>=m||nx<0||ny<0)continue;
                    if(vis[nx][ny]==grid[x][y]){
                        cnt++;
                        if(cnt>=2){
                            return 1;
                        }
                        continue;
                    }
                    if(grid[nx][ny]==grid[x][y]){
                        vis[nx][ny]=grid[nx][ny];
                        q.push({nx,ny});
                    }
                }
            }
            return 0;
        };
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                if(vis[i][j]==0){
                    vis[i][j]=grid[i][j];
                    if(bfs(i,j)){
                        return 1;
                    }
                }
            }
        }
        return 0;
    }
};
```

- 时间复杂度: $O(NM)$
- 空间复杂度: $O(NM)$
