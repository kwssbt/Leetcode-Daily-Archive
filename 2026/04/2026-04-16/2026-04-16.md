## 代码实现 (C++)
```cpp
class Solution {
public:
    vector<int> solveQueries(vector<int>& nums, vector<int>& queries) {
        int n=nums.size();
        for(int i=0;i<n;++i){
            nums.push_back(nums[i]);
        }
        int N=n<<1;
        vector<int>L(N);
        unordered_map<int,int>stL;
        for(int i=0;i<N;++i){
            if(stL.count(nums[i])){
                L[i]=i-stL[nums[i]];
            }
            stL[nums[i]]=i;
        }
        vector<int>R(N);
        unordered_map<int,int>stR;
        for(int i=N-1;i>=0;--i){
            if(stR.count(nums[i])){
                R[i]=stR[nums[i]]-i;
            }
            stR[nums[i]]=i;
        }
        vector<int>ans;
        for(int q:queries){
            int l=L[q+n];
            int r=R[q];
            int dis=min(l,r);
            if(dis==n){
                ans.push_back(-1);
            }
            else{
                ans.push_back(dis);
            }
        }
        return ans;
    }
};
```

- 时间复杂度: $O(N)$
- 空间复杂度: $O(N)$
