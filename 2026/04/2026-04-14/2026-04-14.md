## 记忆化搜索

`dfs(i,j)` 完成前 i 个机器人，且第 i 个机器人由第 j 个工厂处理

---

## 代码实现 (C++)
```cpp
class Solution {
public:
    long long minimumTotalDistance(vector<int>& robot, vector<vector<int>>& factory) {
        sort(robot.begin(),robot.end());
        sort(factory.begin(),factory.end(),[](auto&a,auto&b){return a[0]<b[0];});
        int n=robot.size(),m=factory.size();
        vector mem(n,vector<long long>(m,-1));
        function<long long(int,int)>dfs=[&](int i,int j)->long long{
            if(i==-1){
                return 0;
            }
            if(j==-1){
                return LLONG_MAX/2;
            }
            long long &res=mem[i][j];
            if(res!=-1){
                return res;
            }
            long long sum=0;
            res=dfs(i,j-1);
            for(int k=1;k<=min(factory[j][1],i+1);++k){
                sum+=abs(robot[i-k+1]-factory[j][0]);
                res=min(res,sum+dfs(i-k,j-1));
            }
            return res;
        };
        return dfs(n-1,m-1);
    }
};

```

- 时间复杂度: $O(N*M*K)$
- 空间复杂度: $O(N*M)$
