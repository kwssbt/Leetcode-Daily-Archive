## 代码实现 (C++)
```cpp
class Robot {
    int width,height;
    int x,y;
    int dir;
    int C;
public:
    Robot(int width, int height) {
        this->width=width;
        this->height=height;
        this->x=0;
        this->y=0;
        this->dir=1;
        this->C=(width-1)*2+(height-1)*2;
    }
    
    void step(int num) {
        num=num%C==0?C:num%C;
        if(dir==0){
            if(y+num>=height){
                int t=height-1-y;
                y=height-1;
                dir=3;
                step(num-t);
            }
            else{
                y+=num;
            }
        }
        else if(dir==1){
            if(x+num>=width){
                int t=width-1-x;
                x=width-1;
                dir=0;
                step(num-t);
            }
            else{
                x+=num;
            }
        }
        else if(dir==2){
            if(y-num<0){
                int t=y;
                y=0;
                dir=1;
                step(num-t);
            }
            else{
                y-=num;
            }
        }
        else{
            if(x-num<0){
                int t=x;
                x=0;
                dir=2;
                step(num-t);
            }
            else{
                x-=num;
            }
        }
    }
    
    vector<int> getPos() {
        return vector<int>{x,y};
    }
    
    string getDir() {
        if(dir==0)return "North";
        if(dir==1)return "East";
        if(dir==2)return "South"; 
        return "West";
    }
};

/**
 * Your Robot object will be instantiated and called as such:
 * Robot* obj = new Robot(width, height);
 * obj->step(num);
 * vector<int> param_2 = obj->getPos();
 * string param_3 = obj->getDir();
 */
```
