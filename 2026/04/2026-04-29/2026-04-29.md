## 代码实现 (C++)
```cpp
class Solution {
    const long long inf=1e18;
public:
    long long maximumScore(vector<vector<int>>& grid) {
        int n=grid.size();
        
        vector pre(n+2,vector<long long>(n+2));
        for(int j=1;j<=n;++j){
            for(int i=1;i<=n;++i){
                pre[j][i]=pre[j][i-1]+grid[i-1][j-1];
            }
        }

        vector dp(n+1,vector<vector<long long>>(n+1,vector<long long>(2,-inf)));
        for(int i=1;i<=n;++i){
            for(int j=1;j<=n;++j){
                dp[i][j][1]=pre[i-1][j]+pre[i+1][j];
                if(i>1){
                    for(int k=1;k<=j;++k){
                        long long t=pre[i-1][j]-pre[i-1][k]+pre[i+1][j]-pre[i][k];
                        dp[i][j][1]=max(dp[i][j][1],dp[i-1][k][1]+t);                    
                    }
                    for(int k=j+1;k<=n;++k){
                        long long t=pre[i+1][j]-pre[i][j];
                        dp[i][j][0]=max({dp[i][j][0],dp[i-1][k][0]+t,dp[i-1][k][1]+t});
                    }
                }
                if(i>2){
                    for(int k=1;k<=j;++k){
                        long long t=pre[i-1][j]-pre[i-1][k]+pre[i+1][j];
                        dp[i][j][1]=max({dp[i][j][1],dp[i-2][k][0]+t,dp[i-2][k][1]+t});                    
                    }
                    for(int k=j+1;k<=n;++k){
                        long long t=pre[i+1][j];
                        dp[i][j][1]=max({dp[i][j][1],dp[i-2][k][0]+t,dp[i-2][k][1]+t});
                    }
                }
                if(i>3){
                    for(int k=1;k<=n;++k){
                        long long t=pre[i-1][j]+pre[i+1][j];
                        dp[i][j][1]=max({dp[i][j][1],dp[i-3][k][0]+t,dp[i-3][k][1]+t});                    
                    }
                }
            }
        }

        long long ans=0;
        for(int i=1;i<=n;++i){
            for(int j=1;j<=n;++j){
                ans=max({ans,dp[i][j][0],dp[i][j][1]});
            }
        }
        return ans;
    }
};
```

- 时间复杂度: $O(N^3)$
- 空间复杂度: $O(N^2)$
